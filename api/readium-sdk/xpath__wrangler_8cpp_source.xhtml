39 XPathWrangler::StringList XPathWrangler::Strings(const string& xpath, shared_ptr<xml::Node> node)
40 {
41  StringList strings;
42 
43  xml::XPathEvaluator eval(xml::string(xpath.c_str()), _doc);
44  xml::XPathEvaluator::ObjectType type;
45  for (auto& pair : _namespaces)
46  {
47  eval.RegisterNamespace(pair.first.stl_str(), pair.second.stl_str());
48  }
49 
50  if ( eval.Evaluate((bool(node) ? node : _doc), &type) )
51  {
52  switch ( type )
53  {
54  case xml::XPathEvaluator::ObjectType::String:
55  // a single string
56  strings.emplace_back(eval.StringResult());
57  break;
58  case xml::XPathEvaluator::ObjectType::NodeSet:
59  {
60  xml::NodeSet nodes(eval.NodeSetResult());
61 
62  // a list of strings (I hope)
63  for (shared_ptr<xml::Node> node : nodes)
64  {
65  strings.emplace_back(node->StringValue());
66  }
67  break;
68  }
69  default:
70  break;
71  }
72  }
73 
74  return strings;
75 }
76 xml::NodeSet XPathWrangler::Nodes(const string& xpath, shared_ptr<xml::Node> node)
77 {
78  xml::NodeSet result;

80  xml::XPathEvaluator eval(xml::string(xpath.c_str()), _doc);

82  {
83  eval.RegisterNamespace(item.first.stl_str(), item.second.stl_str());
84  }
85  xml::XPathEvaluator::ObjectType type;
86  if ( eval.Evaluate((bool(node) ? node : _doc), &type) )
87  {
88  if ( type == xml::XPathEvaluator::ObjectType::NodeSet )
89  {
90  result = eval.NodeSetResult();
91  }
92  }
93
