432 // Lo, the std::system:error class's code() method was not marked virtual,
433 // and there was much fannying about.
434 class error_details
435 {
436 private:
437  bool __is_spec_error_;
438  union
439  {
440  const std::system_error* __system_error_;
441  const epub_spec_error* __spec_error_;
442  };
443 
444 public:
445  error_details(const std::system_error& __sysErr)
446  : __is_spec_error_(false), __system_error_(&__sysErr)
447  {}
448  error_details(const epub_spec_error& __specErr)
449  : __is_spec_error_(true), __spec_error_(&__specErr)
450  {}
451  ~error_details() {}
452 
453 private:
454  error_details(const error_details&) _DELETED_;
455  error_details& operator=(const error_details&) _DELETED_;
456 
457 public:
458  bool is_spec_error() const {
459  return __is_spec_error_;
460  }
461  bool is_system_error() const {
462  return !__is_spec_error_;
463  }
464 
465  int code() const {
466  if (__is_spec_error_)
467  return __spec_error_->code().value();
468  else
469  return __system_error_->code().value();
470  }
471  const char* message() const {
472  if (__is_spec_error_)
473  return __spec_error_->what();
474  else
475  return __system_error_->what();
476  }
477 
478  const std::error_category& category() const {
479  if (__is_spec_error_)
480  return __spec_error_->code().category();
481  else
482  return __system_error_->code().category();
483  }
484 
485  EPUBSpec epub_spec() const {
486  if (__is_spec_error_)
487  return __spec_error_->Specification();
488  else
489  throw std::logic_error("Attempt to get an EPUBSpec from a non-epub_spec_error exception");
490  }
491 
492  ViolationSeverity severity() const {
493  if (__is_spec_error_)
494  return __spec_error_->Severity();
495  else
496  throw std::logic_error("Attempt to get a ViolationSeverity from a non-epub_spec_error exception");
497  }
498 
499  EPUBError epub_error_code() const {
500  if (__is_spec_error_)
501  return __spec_error_->SpecErrorCode();
502  else
503  throw std::logic_error("Attempt to get an EPUBError from a non-epub_spec_error exception");
504  }
505 
506  _NORETURN_
507  void throw_error() const {
508  if (__is_spec_error_)
509  throw *__spec_error_;
510  else
511  throw *__system_error_;
512  }
