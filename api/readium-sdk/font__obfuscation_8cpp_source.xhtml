52 void * FontObfuscator::FilterData(FilterContext* context, void *data, size_t len, size_t *outputLen)
53 {
54  FontObfuscationContext* p = dynamic_cast<FontObfuscationContext*>(context);
55  size_t bytesFiltered = p->ProcessedCount();
56 
57  uint8_t *buf = static_cast<uint8_t*>(data);
58  for ( size_t i = 0; i < len && (i + bytesFiltered) < 1040; i++)
59  {
60  // XOR each of the first 1040 bytes of the font with the key, circling around the keybuf
61  buf[i] ^= _key[(i+bytesFiltered)%20];
62  }
63 
64  bytesFiltered += len;
65  p->SetProcessedCount(bytesFiltered);
66  *outputLen = len;
67  return buf;
68 }

70 {

72  std::stringstream ss;

84  auto str = ss.str();
85 
86 #if EPUB_PLATFORM(WIN)
87  HCRYPTPROV csp;
88  if ( ::CryptAcquireContext(&csp, NULL, NULL, PROV_DSS, CRYPT_VERIFYCONTEXT) == FALSE )
89  {
90  _THROW_LAST_ERROR_();
91  }
92 
93  HCRYPTHASH hasher;
94  if ( ::CryptCreateHash(csp, CALG_SHA, 0, 0, &hasher) == FALSE )
95  {
96  ::CryptReleaseContext(csp, 0);
97  _THROW_LAST_ERROR_();
98  }
99 
100  DWORD winerr = NO_ERROR;
101  if ( ::CryptHashData(hasher, reinterpret_cast<const BYTE*>(str.data()), str.length(), 0) == TRUE )
102  {
103  DWORD len = KeySize;
104  if ( ::CryptGetHashParam(hasher, HP_HASHVAL, _key, &len, 0) == FALSE )
105  winerr = ::GetLastError();
106  }
107  else
108  {
109  winerr = ::GetLastError();
110  }
111 
112  ::CryptDestroyHash(hasher);
113  ::CryptReleaseContext(csp, 0);
114 
115  if ( winerr != NO_ERROR )
116  _THROW_WIN_ERROR_(winerr);
117 #elif EPUB_PLATFORM(WINRT)
118  auto byteArray = ArrayReference<byte>(reinterpret_cast<byte*>(const_cast<char*>(str.data())), str.length());
119  auto inBuf = CryptographicBuffer::CreateFromByteArray(byteArray);
120  auto keyBuf = HashAlgorithmProvider::OpenAlgorithm(HashAlgorithmNames::Sha1)->HashData(inBuf);
121 
122  Array<byte>^ outArray = nullptr;
123  CryptographicBuffer::CopyToByteArray(keyBuf, &outArray); // creates a new Array<byte>^ and returns it by reference
124  memcpy_s(_key, KeySize, outArray->Data, outArray->Length);
125 #else
126  // hash the accumulated string (using OpenSSL syntax for portability)
127  SHA_CTX ctx;
128  SHA1_Init(&ctx);
129  SHA1_Update(&ctx, str.data(), str.length());
130  SHA1_Final(_key, &ctx);
131 #endif
132  return true;
133 }
